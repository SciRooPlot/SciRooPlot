60// some preprocessor macros for logging, printing and debugging
61#define DEBUG(s, ...) \
62 { \
63 fmt::print(stderr, CYAN_ "[ DEBU ]" _END " "); \
64 fmt::print(stderr, s, ##__VA_ARGS__); \
65 fmt::print(stderr, "\n"); \
66 }
67#define WARNING(s, ...) \
68 { \
69 fmt::print(stderr, YELLOW_ "[ WARN ]" _END " "); \
70 fmt::print(stderr, s, ##__VA_ARGS__); \
71 fmt::print(stderr, "\n"); \
72 }
73#define ERROR(s, ...) \
74 { \
75 fmt::print(stderr, RED_ "[ ERR ]" _END " "); \
76 fmt::print(stderr, s, ##__VA_ARGS__); \
77 fmt::print(stderr, "\n"); \
78 }
79#define LOG(s, ...) \
80 { \
81 fmt::print(GREEN_ "[ LOG ]" _END " "); \
82 fmt::print(s, ##__VA_ARGS__); \
83 fmt::print("\n"); \
84 }
85#define INFO(s, ...) \
86 { \
87 fmt::print("[ INFO ] "); \
88 fmt::print(s, ##__VA_ARGS__); \
89 fmt::print("\n"); \
90 }
91#define PRINT(s, ...) \
92 { \
93 fmt::print(stderr, " | "); \
94 fmt::print(s, ##__VA_ARGS__); \
95 fmt::print("\n"); \
96 }
97#define PRINT_INLINE(s, ...) \
98 { \
99 fmt::print(s, ##__VA_ARGS__); \
100 }
101#define PRINT_SEPARATOR \
102 { \
103 PRINT(fmt::format("{:-<{}}", "-", 60)); \
104 }
105#define HERE \
106 { \
107 fmt::print("[ ---> ] Line {} in function {} ({})", __LINE__, __FUNCTION__, __FILE__); \
108 fmt::print("\n"); \
109 }
